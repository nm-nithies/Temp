def get_dtype(dtype):
    """Returns the corresponding NumPy dtype for a given ONNX Runtime dtype
    """
    numpy_dtype =  tensor_dtype_to_np_dtype(dtype)
    
    logger.warning(
        f"[WARNING] The dtype '{dtype}' is unsupported in NumPy representation."
        "Treating as float32 as a trial.")
    numpy_dtype = np.float32
    return numpy_dtype
