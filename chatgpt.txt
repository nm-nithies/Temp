  only_float = any('.' in v or 'e' in v.lower() for v in val_list)
                only_int = all(v.isdigit() or (v.startswith('-') and v[1:].isdigit()) for v in val_list)

                if only_float and only_int:
                    logger.error(f"[ERROR] Mixed int, float types and are not supported: {val_list}")
                    sys.exit(-1)
