parser.add_argument(
    '--replace_input_with_constant',
    nargs='+',
    help=(
        "Replaces one or more input nodes in the model with constant values. "
        "Each input should be followed by its replacement value: either a scalar (int or float) or "
        "an absolute path to a .npy file containing the tensor data. "
        "The provided values will be validated for type and shape consistency against the original model inputs. "
        "Example usage:\n"
        "  --replace_input_with_constant input1 1.0\n"
        "  --replace_input_with_constant input2 input2.npy\n"
        "  --replace_input_with_constant input1 1.0 input2 input2.npy\n"
        "\n"
        "Note:\n"
        "- Scalars can be broadcasted to the expected shape if allowed.\n"
        "- .npy files must contain data matching the input's expected dtype and shape.\n"
        "- This is useful for freezing inputs or simplifying models for inference or deployment."
    )
)
