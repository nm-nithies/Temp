
parser.add_argument(
    "--legalize_replace_input_with_constant",
    type=str,
    nargs="+",
    action="append",
    default=None,
    help=(
        "Replaces one or more input nodes in the model with constant values. "
        "Each input must be followed by its replacement value: either a scalar (int or float) or "
        "an absolute path to a .npy file containing the full tensor data. "
        "Scalar values are only allowed for 1D input shapes and will be broadcasted to match the full 1D shape. "
        "For multi-dimensional inputs (e.g., 2D, 4D), a .npy file must be used. "
        "All values will be validated for data type and shape consistency. "
        "Example usage:\n"
        "  --legalize_replace_input_with_constant input1 1.0\n"
        "  --legalize_replace_input_with_constant input2 input2.npy\n"
        "  --legalize_replace_input_with_constant input1 1 --legalize_replace_input_with_constant input2 input2.npy"
    )
)
