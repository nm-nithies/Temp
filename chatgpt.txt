def get_dtype(dtype):
    """
    Returns the corresponding NumPy dtype for a given tensor dtype.
    """
    numpy_dtype = tensor_dtype_to_np_dtype(dtype)
    if numpy_dtype is not None:
        return numpy_dtype
    else:
        logger.warning(
           f"[WARNING] Dtype '{dtype}' is not supported. Defaulting to float32."
        )
        return np.float32
