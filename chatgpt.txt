def get_removed_input_node_names(legalize_replace_input_with_constant, input_data_types):
    """This method extracts the corresponding input node names along with their constant values
       from inputs provided with the --legalize_replace_input_with_constant argument.
    """
    result = {}
    i = 0
    while i < len(legalize_replace_input_with_constant):
        input_name = legalize_replace_input_with_constant[i]
        val = legalize_replace_input_with_constant[i + 1]
        if val.endswith('.npy'):
            loaded_val = np.load(val)
        else:
            val = int(val)
            loaded_val = np.array([val], dtype=get_dtype(input_data_types[input_name]))

        result[input_name] = loaded_val
        i += 2

    return result
