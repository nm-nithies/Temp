def get_removed_input_node_names(legalize_replace_input_with_constant, input_data_types):
    """
    Extracts input node names and constant values from the given input pairs.
    
    Args:
        legalize_replace_input_with_constant (list of [str, str]): e.g., [['input1', '1.0'], ['input2', 'input2.npy']]
        input_data_types (dict): Mapping of input names to expected data types

    Returns:
        dict: Mapping of input_name -> constant value (np.ndarray)
    """
    result = {}

    for pair in legalize_replace_input_with_constant:
        if len(pair) != 2:
            raise ValueError(f"Each item must be a pair of (input_name, value_or_file), got: {pair}")

        input_name, val = pair

        if val.endswith('.npy'):
            loaded_val = np.load(val)
        else:
            val = int(val)  # Optional: enhance this if float values are possible
            dtype = get_dtype(input_data_types[input_name])
            loaded_val = np.array([val], dtype=dtype)

        result[input_name] = loaded_val

    return result
