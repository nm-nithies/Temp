@script(local)
def RMSNormalization(X, W, axis, epsilon: float):
    two = op.Constant(value_int=2)
    eps_conf = op.Constant(value_float=epsilon)
    # axis = op.Constant(value_ints=axis)
    x_sq = op.Pow(X, two)

    mean = op.ReduceMean(x_sq, axes=axis, keepdims=1)
    var = op.Add(mean, eps_conf)
    rms = op.Sqrt(var)
    inv_rms = op.Reciprocal(rms)

    Y = op.Mul(op.Mul(X, inv_rms), W)
    return Y
