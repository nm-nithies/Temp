def get_constants(curr_layer_output, matmul_inits, add_inits):
    """Append and return all parallel MatMul and Add constants in a list
    """
    mul_const = []
    add_const = []
    for layer in curr_layer_output:
        if layer in matmul_inits:
            mul_const.append(matmul_inits[layer])
            add_const.append(add_inits[layer])

    return mul_const, add_const
